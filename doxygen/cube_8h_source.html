304  // 1. Convert map coordinates to view-based coordinates
305  coords_nd<uint32_t, 3> s = _st_ref->cube_coords(p);
306 
307  // 2. Find out which chunk contains the integer view coordinates
308  id += cumprod * (s[2] / _chunk_size[2]);
309  cumprod *= (uint32_t)std::ceil(((double)_st_ref->nx()) / ((double)_chunk_size[2]));
310  id += cumprod * (s[1] / _chunk_size[1]);
311  cumprod *= (uint32_t)std::ceil(((double)(_st_ref->ny()) / ((double)_chunk_size[1])));
312  id += cumprod * (s[0] / _chunk_size[0]);
313  //cumprod *= (uint32_t)std::ceil(((double)(_c->view().nt()) / ((double)_size_t);
314 
315  return id;

323  bounds_nd<uint32_t, 3> chunk_limits(chunk_coordinate_tyx c) const {
324  cube_coordinate_tyx out_vcoords_low;
325  cube_coordinate_tyx out_vcoords_high;
326 
327  out_vcoords_low[0] = c[0] * _chunk_size[0];
328  out_vcoords_low[1] = c[1] * _chunk_size[1];
329  out_vcoords_low[2] = c[2] * _chunk_size[2];
330  out_vcoords_high[0] = out_vcoords_low[0] + _chunk_size[0] - 1;
331  out_vcoords_high[1] = out_vcoords_low[1] + _chunk_size[1] - 1;
332  out_vcoords_high[2] = out_vcoords_low[2] + _chunk_size[2] - 1;
333 
334  // Shrink to view
335  if (out_vcoords_high[0] < 0)
336  out_vcoords_high[0] = 0;
337  else if (out_vcoords_high[0] >= _st_ref->nt())
338  out_vcoords_high[0] = _st_ref->nt() - 1;
339  if (out_vcoords_low[0] < 0)
340  out_vcoords_low[0] = 0;
341  else if (out_vcoords_low[0] >= _st_ref->nt())
342  out_vcoords_low[0] = _st_ref->nt() - 1;
343 
344  if (out_vcoords_high[1] < 0)
345  out_vcoords_high[1] = 0;
346  else if (out_vcoords_high[1] >= _st_ref->ny())
347  out_vcoords_high[1] = _st_ref->ny() - 1;
348  if (out_vcoords_low[1] < 0)
349  out_vcoords_low[1] = 0;
350  else if (out_vcoords_low[1] >= _st_ref->ny())
351  out_vcoords_low[1] = _st_ref->ny() - 1;
352 
353  if (out_vcoords_high[2] < 0)

373  bounds_nd<uint32_t, 3> chunk_limits(chunkid_t id) const {
374  coords_nd<uint32_t, 3> out_vcoords_low;
375  coords_nd<uint32_t, 3> out_vcoords_high;
376 
377  // Transform to global coordinates based on chunk id
378  uint32_t cumprod = 1;
379  int32_t n;
380 
381  n = (uint32_t)std::ceil(((double)(_st_ref->nx())) / ((double)_chunk_size[2]));
382  out_vcoords_low[2] = (id / cumprod) % n;
383  out_vcoords_low[2] *= _chunk_size[2];
384  out_vcoords_high[2] = out_vcoords_low[2] + _chunk_size[2] - 1;
385  cumprod *= n;
386 
387  n = (uint32_t)std::ceil(((double)(_st_ref->ny())) / ((double)_chunk_size[1]));
388  out_vcoords_low[1] = (id / cumprod) % n;
389  out_vcoords_low[1] *= _chunk_size[1];
390  out_vcoords_high[1] = out_vcoords_low[1] + _chunk_size[1] - 1;
391  cumprod *= n;
392 
393  n = (uint32_t)std::ceil(((double)(_st_ref->nt())) / ((double)_chunk_size[0]));
394  out_vcoords_low[0] = (id / cumprod) % n;
395  out_vcoords_low[0] *= _chunk_size[0];
396  out_vcoords_high[0] = out_vcoords_low[0] + _chunk_size[0] - 1;
397  cumprod *= n;
398 
399  // Shrink to view
400  if (out_vcoords_high[0] < 0)
401  out_vcoords_high[0] = 0;
402  else if (out_vcoords_high[0] >= _st_ref->nt())
403  out_vcoords_high[0] = _st_ref->nt() - 1;
404  if (out_vcoords_low[0] < 0)
405  out_vcoords_low[0] = 0;
406  else if (out_vcoords_low[0] >= _st_ref->nt())
407  out_vcoords_low[0] = _st_ref->nt() - 1;
408 
409  if (out_vcoords_high[1] < 0)
410  out_vcoords_high[1] = 0;
411  else if (out_vcoords_high[1] >= _st_ref->ny())
412  out_vcoords_high[1] = _st_ref->ny() - 1;
413  if (out_vcoords_low[1] < 0)
414  out_vcoords_low[1] = 0;
415  else if (out_vcoords_low[1] >= _st_ref->ny())
416  out_vcoords_low[1] = _st_ref->ny() - 1;
417 
418  if (out_vcoords_high[2] < 0)
419  out_vcoords_high[2] = 0;
420  else if (out_vcoords_high[2] >= _st_ref->nx())
421  out_vcoords_high[2] = _st_ref->nx() - 1;
422  if (out_vcoords_low[2] < 0)
423  out_vcoords_low[2] = 0;
424  else if (out_vcoords_low[2] >= _st_ref->nx())
425  out_vcoords_low[2] = _st_ref->nx() - 1;
426 
427  bounds_nd<uint32_t, 3> out;

732  std::shared_ptr<cube> p = _succ[i].lock();
