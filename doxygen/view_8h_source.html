241  _nx = (uint32_t)std::ceil((_win.right - _win.left) / x);

321  uint32_t nt() {
322  if (_t1 == _t0) return 1;
323  duration d = (_t1 - _t0) + 1;
324  return (d % _dt == 0) ? d / _dt : (1 + (d / _dt));
325  }

331  void nt(uint32_t n) {
332  duration d = (_t1 - _t0) + 1;
333  dt().dt_interval = (int32_t)std::ceil((double)d.dt_interval / (double)n);
334  if (nt() == n - 1) { // in some cases (e.g. d == 9M, n==4), we must extend the temporal extent of the view
335  _t1 = _t1 + dt();
336  }
337  assert(nt() == n);
338  }

418  coords_nd<uint32_t, 3> cube_coords(coords_st p) {

421  s[1] = (uint32_t)((p.s.y - _win.bottom) / dy());
