221 struct median_reducer : public reducer {
222  void init(std::shared_ptr<chunk_data> a) override {
223  //_m_buckets = (std::vector<double> *)std::calloc(a->size()[0] * a->size()[2] * a->size()[3], sizeof(std::vector<double>));
224  _m_buckets.resize(a->size()[0] * a->size()[2] * a->size()[3], std::vector<double>());
225  }
226 
227  void combine(std::shared_ptr<chunk_data> a, std::shared_ptr<chunk_data> b) override {
228  for (uint16_t ib = 0; ib < a->size()[0]; ++ib) {
229  for (uint32_t it = 0; it < b->size()[1]; ++it) {
230  for (uint32_t ixy = 0; ixy < b->size()[2] * b->size()[3]; ++ixy) {
231  double v = ((double *)b->buf())[ib * b->size()[1] * b->size()[2] * b->size()[3] + it * b->size()[2] * b->size()[3] + ixy];
232  if (!std::isnan(v)) {
233  _m_buckets[ib * a->size()[2] * a->size()[3] + ixy].push_back(v);
234  }
235  }
236  }
237  }
238  }
239 
240  void finalize(std::shared_ptr<chunk_data> a) override {
241  // divide by count;
242  for (uint32_t ibxy = 0; ibxy < a->size()[0] * a->size()[2] * a->size()[3]; ++ibxy) {
243  std::vector<double> &list = _m_buckets[ibxy];
244  std::sort(list.begin(), list.end());
245  if (list.size() == 0) {
246  ((double *)a->buf())[ibxy] = NAN;
247  } else if (list.size() % 2 == 1) {
248  ((double *)a->buf())[ibxy] = list[list.size() / 2];
249  } else {
250  ((double *)a->buf())[ibxy] = (list[list.size() / 2] + list[list.size() / 2 - 1]) / ((double)2);
251  }
252  }
253  }
254 
255  private:
256  std::vector<std::vector<double>> _m_buckets;

262 struct var_reducer : public reducer {
263  void init(std::shared_ptr<chunk_data> a) override {
264  _count = (uint32_t *)std::calloc(a->size()[0] * a->size()[2] * a->size()[3], sizeof(uint32_t));
265  _mean = (double *)std::calloc(a->size()[0] * a->size()[2] * a->size()[3], sizeof(uint32_t));
266  for (uint32_t ibxy = 0; ibxy < a->size()[0] * a->size()[2] * a->size()[3]; ++ibxy) {
267  _count[ibxy] = 0;
268  _mean[ibxy] = 0;
269  ((double *)a->buf())[ibxy] = 0;
270  }
271  }
272 
273  void combine(std::shared_ptr<chunk_data> a, std::shared_ptr<chunk_data> b) override {
274  for (uint16_t ib = 0; ib < a->size()[0]; ++ib) {
275  for (uint32_t it = 0; it < b->size()[1]; ++it) {
276  for (uint32_t ixy = 0; ixy < b->size()[2] * b->size()[3]; ++ixy) {
277  double &v = ((double *)b->buf())[ib * b->size()[1] * b->size()[2] * b->size()[3] + it * b->size()[2] * b->size()[3] + ixy];
278  if (!std::isnan(v)) {
279  double &mean = _mean[ib * a->size()[2] * a->size()[3] + ixy];
280  uint32_t &count = _count[ib * a->size()[2] * a->size()[3] + ixy];
281  ++count;
282  double delta = v - mean;
283  mean += delta / count;
284  ((double *)a->buf())[ib * a->size()[1] * a->size()[2] * a->size()[3] + ixy] += delta * (v - mean);
285  }
286  }
287  }
288  }
289  }
290 
291  virtual void finalize(std::shared_ptr<chunk_data> a) override {
292  // divide by count - 1;
293  for (uint32_t ibxy = 0; ibxy < a->size()[0] * a->size()[2] * a->size()[3]; ++ibxy) {
294  ((double *)a->buf())[ibxy] = _count[ibxy] > 1 ? ((double *)a->buf())[ibxy] / (_count[ibxy] - 1) : NAN;
295  }
296  std::free(_count);
297  std::free(_mean);
298  }
299 
300  protected:
301  uint32_t *_count;
302  double *_mean;
