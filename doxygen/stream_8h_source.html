50  stream_cube(std::shared_ptr<cube> in_cube, std::string cmd, std::string log_output = "") : cube(std::make_shared<cube_st_reference>(*(in_cube->st_reference()))), _in_cube(in_cube), _cmd(cmd), _log_output(log_output), _keep_input_nt(false), _keep_input_ny(false), _keep_input_nx(false) { // it is important to duplicate st reference here, otherwise changes will affect input cube as well
51  // Test CMD and find out what size comes out.
52  cube_size_tyx tmp = _in_cube->chunk_size(0);
53  cube_size_btyx csize_in = {_in_cube->bands().count(), tmp[0], tmp[1], tmp[2]};
54 
55  // do not read original chunk data (which can be expensive) but simply stream a dummy chunk with proper size here
56  std::shared_ptr<chunk_data> dummy_chunk = std::make_shared<chunk_data>();
57  dummy_chunk->size(csize_in);
58  dummy_chunk->buf(std::calloc(csize_in[0] * csize_in[1] * csize_in[2] * csize_in[3], sizeof(double)));
59 
60  std::shared_ptr<chunk_data> c0 = stream_chunk_stdin(dummy_chunk);
61 
62  for (uint16_t ib = 0; ib < c0->size()[0]; ++ib) {
63  band b("band" + std::to_string(ib + 1));
64  b.no_data_value = std::to_string(NAN);
65  b.type = "float64";
66  _bands.add(b);
67  }
68 
69  _chunk_size = {c0->size()[1], c0->size()[2], c0->size()[3]};
70 
71  // Make an optimistic guess
72  if (c0->size()[1] == 1) {
73  _st_ref->nt(in_cube->count_chunks_t());
74  } else if (c0->size()[1] == csize_in[1]) {
75  _keep_input_nt = true;
76  _st_ref->nt(in_cube->size()[1]);
77  } else
78  throw std::string("ERROR in stream_cube::stream_cube(): could not derive size of result cube");
79 
80  if (c0->size()[2] == 1) {
81  _st_ref->ny() = in_cube->count_chunks_y();
82  } else if (c0->size()[2] == csize_in[2]) {
83  _keep_input_ny = true;
84  _st_ref->ny() = in_cube->size()[2];
85  } else
86  throw std::string("ERROR in stream_cube::stream_cube(): could not derive size of result cube");
87 
88  if (c0->size()[3] == 1) {
89  _st_ref->nx() = in_cube->count_chunks_x();
90  } else if (c0->size()[3] == csize_in[3]) {
91  _keep_input_nx = true;
92  _st_ref->nx() = in_cube->size()[3];
93  } else
94  throw std::string("ERROR in stream_cube::stream_cube(): could not derive size of result cube");
95  }
