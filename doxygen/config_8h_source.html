146  void gdalcubes_init() {
147  curl_global_init(CURL_GLOBAL_ALL);
148  GDALAllRegister();
149  GDALSetCacheMax(_gdal_cache_max);
150  CPLSetConfigOption("GDAL_PAM_ENABLED", "NO"); // avoid aux files for PNG tiles
151  curl_global_init(CURL_GLOBAL_ALL);
152  CPLSetConfigOption("GDAL_NUM_THREADS", std::to_string(_gdal_num_threads).c_str());
153  srand(time(NULL));
154  CPLSetErrorHandler(CPLQuietErrorHandler);
155 
156  // Add default locations where to look for collection format presets
157 
158  if (std::getenv("GDALCUBES_DATA_DIR") != NULL) {
159  if (filesystem::exists(std::getenv("GDALCUBES_DATA_DIR"))) {
160  config::instance()->add_collection_format_preset_dir(std::getenv("GDALCUBES_DATA_DIR"));
161  }
162  }
163  if (std::getenv("AllUsersProfile") != NULL) {
164  // Windows default location
165  std::string p = filesystem::join(filesystem::join(std::getenv("AllUsersProfile"), "gdalcubes"), "formats");
166  if (filesystem::exists(p)) {
167  config::instance()->add_collection_format_preset_dir(p);
168  }
169  }
170  if (std::getenv("HOME") != NULL) {
171  std::string p = filesystem::join(filesystem::join(std::getenv("HOME"), ".gdalcubes"), "formats");
172  if (filesystem::exists(p)) {
173  config::instance()->add_collection_format_preset_dir(p);
174  }
175  }
176  if (std::getenv("HOMEPATH") != NULL && getenv("HOMEDRIVE") != NULL) {
177  std::string p = filesystem::join(filesystem::join(std::getenv("HOMEPATH"), ".gdalcubes"), "formats");
178  if (filesystem::exists(p)) {
179  config::instance()->add_collection_format_preset_dir(p);
180  }
181  }
182 
183  const std::vector<std::string> candidate_dirs =
184  {"/usr/lib/gdalcubes/formats"};
185 
186  for (uint16_t i = 0; i < candidate_dirs.size(); ++i) {
187  if (filesystem::exists(candidate_dirs[i])) {
188  config::instance()->add_collection_format_preset_dir(candidate_dirs[i]);
189  }
190  }
