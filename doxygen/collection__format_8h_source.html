48  static std::map<std::string, std::string> list_presets() {
49  std::map<std::string, std::string> out;
50 
51  std::vector<std::string> dirs = config::instance()->get_collection_format_preset_dirs();
52 
53  // do not use uint here because of descending iteration
54  for (int i = dirs.size() - 1; i >= 0; --i) {
55  if (!filesystem::exists(dirs[i])) {
56  continue;
57  }
58 
59  filesystem::iterate_directory(dirs[i], [&out](const std::string& p) {
60  if (filesystem::is_regular_file(p) && filesystem::extension(p) == "json") {
61  if (out.find(filesystem::stem(p)) == out.end())
62  out.insert(std::pair<std::string, std::string>(filesystem::stem(p), filesystem::make_absolute(p)));
63  }
64  });
65  }
66  return out;
67  }

75  void load_file(std::string filename) {
76  _j.clear();
77 
78  if ((!filesystem::exists(filename) && !filesystem::is_absolute(filename) && filesystem::parent(filename).empty()) || filesystem::is_directory(filename)) {
79  // simple filename without directories
80  GCBS_DEBUG("Couldn't find collection format '" + filename + "', looking for a preset with the same name");
81  std::map<std::string, std::string> preset_formats = list_presets();
82  if (preset_formats.find(filesystem::stem(filename)) != preset_formats.end()) {
83  filename = preset_formats[filesystem::stem(filename)];
84  GCBS_DEBUG("Found collection format preset at '" + filename + "'");
85  }
86  }
87  if (!filesystem::exists(filename) || filesystem::is_directory(filename))
88  throw std::string("ERROR in collection_format::load_file(): image collection format file does not exist.");
89 
90  std::ifstream i(filename);
91  i >> _j;
92  }
